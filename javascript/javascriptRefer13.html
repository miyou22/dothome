06. 구구단 만들기 

{
            //2~9 for( var i=2;, i<=9; i++ )
            //1~9 for( var i=2;, i<=9; i++ )
            //num1 * num2 = sum

            //2 * 1 = 2;          3 * 1 = 3;                    
            //2 * 2 = 4;          3 * 2 = 6;
            //2 * 3 = 6;          3 * 3 = 9;
            //2 * 4 = 8;          ..........
            //2 * 5 = 10;         9 * 1 = 9;
            //2 * 6 = 12;         9 * 2 = 18;    
            //2 * 7 = 14;
            //2 * 8 = 16;
            //2 * 9 = 18;
    document.write("*** 06. 구구단 만들기 ***");
    for( var i=2; i<=9; i++){
        for( var j=1; j<=9; j++){
            var sum = i * j
            document.write(i + " * " + j + " = " + sum, "");
        }
    }
}

07. 테이블 만들기

{
    document.write("*** 07. 테이블 만들기 ***
");
    let num = 1;
    let table = "";

        for( let i=1; i<=10; i++ ){
            table +="";
            for( let j=1; j<=10; j++ ){
                table += "";
                num ++;
             }
            table +="";
         }
        table +="
"+ num +"
";
        document.write(table);
}

08. 테이블 만들기2 

{

09. 테이블 만들기3 - 짝수 

{
    document.write("*** 08. 테이블 만들기 ***");
    let num = 2;
    let table = "";

        for( let i=1; i<=10; i++ ){
            table +="";

             }
