19.재귀 함수 : 동기 / 비동기

    document.write("*** 19. 콜백 함수 ***");

    function funA(){
        document.write("funA가 실행되었습니다.");
        console.log("funA가 실행되었습니다.");
    }
    function funB(){
        document.write("funB가 실행되었습니다.");
        console.log("funB가 실행되었습니다.");
    }
    
    funB();
    funA();

    function funC(){
        setTimeout(function(){
        console.log("funC가 실행되었습니다.");
    }, 1000);
    function funD(){
        console.log("funD가 실행되었습니다.");
    }
    funC();
    funD();
    }

  
  function funE(callback){
    setTimeout(() => {
    console.log("funE가 실행되었습니다.");
    callback();
  }, 1000);
  }
  function funF(){
    console.log("funF가 실행되었습니다. ");
  }
  funE(funF);
}

26. 함수 유형 : 객체 생성자 함수 

{
    document.write("*** 26. 함수 유형 : 객체 생성자 함수 ***");

    function func(num, name, job){
        this.num = num;
        this.name = name;
        this.job = job;

        this.result = function(){
           document.write(this.num + ". 내이름은" + this.name + "이며, 직업은 " + this.job + "입니다.");
        }

    }
    const info1 = new func("1","서비","웹퍼블리셔");   //인스턴스 생성 -- object,array,date와 다르게 기능을 본인이 만듬.
    const info2 = new func("2","대서비","프론트엔드 개발자");   //인스턴스 생성 -- object,array,date와 다르게 기능을 본인이 만듬.

    info1.result();
    info2.result();
}

27. 함수 유형 : 프로토타입 함수

         this.name = name;

            document.write(this.num + ". 내이름은" + this.name + "이며, 직업은 " + this.job + "입니다.");
         }

    const info2 = new func("2","대서비","프론트엔드 개발자");

    info1.result();
    info2.result();
}
