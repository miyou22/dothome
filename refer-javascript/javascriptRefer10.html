02. 속성 메서드 : toggleClass()

이미지를 클릭하면 모든 이미지에 클래스(active)를 추가/삭제 합니다. 
{
    //모든 이미지 선택
        document.querySelectorAll(".ex2 div").forEach((el)=>{
            el.addEventListener("click", function(){
                document.querySelectorAll(".ex2 div").forEach((el) => {
                    el.classList.toggle("active");
                })
            });
        });
    //클릭한 이미지만 선택
    document.querySelectorAll(".ex2 div").forEach((el) => {
        el.addEventListener("click", function(){ //화살표함수는 this를 인식하지 못한다.
            this.classList.toggle("active");
        });
    });
}
