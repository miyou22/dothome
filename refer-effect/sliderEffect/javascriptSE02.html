    const sliderWrap = document.querySelector(".slider-wrap")  //document안에서 찾기
    const sliderImg = sliderWrap.querySelector(".slider-img")    //sliderWrap 안에서 찾기
    const sliderInner = sliderImg.querySelector(".slider-inner") 
    const slider = sliderInner.querySelectorAll(".slider")
 
    let currentIndex = 0;
    let sliderCount = slider.length;
    let sliderInterval = 2000;
 
    setInterval (()=>{
         currentIndex < sliderCount-1 ? currentIndex++ : currentIndex=0;
 
         let sliderPosition = currentIndex * -800 + "px";
         sliderInner.style.left = sliderPosition;
         sliderInner.style.transition = "all 0.6s";
 
 
         console.log(currentIndex)
    }, sliderInterval)
